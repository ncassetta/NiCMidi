71 // Used by the MIDIInDriver to keep track of incoming messages. It holds
72 // a MIDIMessage, a timestamp in milliseconds (from the start of the timer) and the id of
73 // the MIDI in port from which the message comes.
74 struct MIDIRawMessage {
75  MIDIRawMessage() : timestamp(0), port(0) {}
76  MIDIRawMessage(const MIDIMessage& m, tMsecs t, int p) :
77  msg(m), timestamp(t), port(p) {}
78  MIDIMessage msg; // The MIDI Message received from the port
79  tMsecs timestamp; // The absolute time in msecs
80  int port; // The id of the MIDI in port which received the message
81 };

85 // This is a queue of MIDIRawMessage
86 class MIDIRawMessageQueue {
87  public:
88  // The constructor creates a queue of the given size. When the queue is full, older messages
89  // are pulled from the queue.
90 
91  MIDIRawMessageQueue(unsigned int size) :
92  next_in(0), next_out(0), buffer(size) {}
93  // The destructor deletes all the MIDIRawMessage objects actually contained in the queue.
94  virtual ~MIDIRawMessageQueue() {}
95  // Empties the queue and turns into a NoOp all the messages contained.
96  void Reset();
97  // Quickly empties the queue acting only on in and out indexes.
98  void Flush() { next_out = next_in; }
99  // Adds the given MIDIRawMessage as the last element in the queue. If the queue was full
100  // the first message is pulled out.
101  void PutMessage(const MIDIRawMessage& msg);
102  // Gets the first MIDIRawMessage in the queue, pulling it out. It returns a reference
103  // to a static copy, which is valid until the next call to the function.
104  MIDIRawMessage& GetMessage();
105  // Gets the n-th MIDIRawMessage in the queue, without pulling it out. It returns a
106  // direct reference to the message (and it's so faster than GetMessage(), which copies it)
107  // which is valid until an operation on the queue is done. If the queue has an actual
108  // size lesser than _n_, returns a NoOp message.
109  MIDIRawMessage& ReadMessage(unsigned int n);
110  // Returns *true* is the queue is empty.
111  bool IsEmpty() const { return next_in == next_out; }
112  // Returns *true* if the queue has reached its max size (you can however add other messages,
113  // deleting the older ones.
114  bool IsFull() const { return ((next_in + 1) % buffer.size()) == next_out; }
115  // Returns the actual length of the queue.
116  unsigned int GetLength() const { return (next_in - next_out) % buffer.size(); }
117 
118  protected:
119  unsigned int next_in;
120  unsigned int next_out;
121  std::vector<MIDIRawMessage> buffer;
122 };
