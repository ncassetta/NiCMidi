66// Used by the MIDIInDriver to keep track of incoming messages. It holds
67// a MIDIMessage, a timestamp in milliseconds (from the start of the timer) and the id of
68// the MIDI in port from which the message comes.
69struct MIDIRawMessage {
70 MIDIRawMessage() : timestamp(0), port(0) {}
71 MIDIRawMessage(const MIDIMessage& m, tMsecs t, int p) :
72 msg(m), timestamp(t), port(p) {}
73 MIDIMessage msg; // The MIDI Message received from the port
74 tMsecs timestamp; // The absolute time in msecs
75 int port; // The id of the MIDI in port which received the message
76};

80// This is a queue of MIDIRawMessage
81class MIDIRawMessageQueue {
82 public:
83 // The constructor creates a queue of the given size. When the queue is full, older messages
84 // are pulled from the queue.
85
86 MIDIRawMessageQueue(unsigned int size) :
87 next_in(0), next_out(0), buffer(size) {}
88 // The destructor deletes all the MIDIRawMessage objects actually contained in the queue.
89 virtual ~MIDIRawMessageQueue() {}
90 // Empties the queue and turns into a NoOp all the messages contained.
91 void Reset();
92 // Quickly empties the queue acting only on in and out indexes.
93 void Flush() { next_out = next_in; }
94 // Adds the given MIDIRawMessage as the last element in the queue. If the queue was full
95 // the first message is pulled out.
96 void PutMessage(const MIDIRawMessage& msg);
97 // Gets the first MIDIRawMessage in the queue, pulling it out. It returns a reference
98 // to a static copy, which is valid until the next call to the function.
99 MIDIRawMessage& GetMessage();
100 // Gets the n-th MIDIRawMessage in the queue, without pulling it out. It returns a
101 // direct reference to the message (and it's so faster than GetMessage(), which copies it)
102 // which is valid until an operation on the queue is done. If the queue has an actual
103 // size lesser than _n_, returns a NoOp message.
104 MIDIRawMessage& ReadMessage(unsigned int n);
105 // Returns *true* is the queue is empty.
106 bool IsEmpty() const { return next_in == next_out; }
107 // Returns *true* if the queue has reached its max size (you can however add other messages,
108 // deleting the older ones.
109 bool IsFull() const { return ((next_in + 1) % buffer.size()) == next_out; }
110 // Returns the actual length of the queue.
111 unsigned int GetLength() const { return (next_in - next_out) % buffer.size(); }
112
113 protected:
114 unsigned int next_in;
115 unsigned int next_out;
116 std::vector<MIDIRawMessage> buffer;
117};
