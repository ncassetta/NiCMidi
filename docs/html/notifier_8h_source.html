54 class MIDISequencerGUIEvent {
55  public:
57  MIDISequencerGUIEvent() { bits = 0; }
59  MIDISequencerGUIEvent(unsigned long bits_) : bits(bits_) {}
61  MIDISequencerGUIEvent( int group, int subgroup, int item ) {
62  bits = ((group&0xff)<<24) | ((subgroup&0xfff)<<12) | (item&0xfff); }
63  // leave unchanged! overloading trouble, too many ctors
65  MIDISequencerGUIEvent(const MIDISequencerGUIEvent &ev) : bits(ev.bits) {}
67  operator unsigned long () const { return bits; }
69  int GetGroup() const { return (int)((bits>>24)&0xff); }
72  int GetSubGroup() const { return (int)((bits>>12)&0xfff); }
74  int GetItem() const { return (int)((bits>>0)&0xfff); }
76  void SetEvent( int group, int subgroup = 0, int item = 0 ) {
77  bits = ((group&0xff)<<24) | ((subgroup&0xfff)<<12) | (item&0xfff); }
